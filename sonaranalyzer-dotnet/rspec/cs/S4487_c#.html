Private fields only used to store values without reading them later is a case of dead store. So changing the value of such field is useless and most probably indicates a serious error in the code.

Noncompliant Code Example

public class Cuboid
{
  private readonly int length;
  private int Breadth { get; set; } // Breadth is written but never read
  private readonly int height;  // height is written but never read

  public Cuboid(int length, int breadth, int height)
  {
    this.length = length;
    this.Breadth = breadth;
    this.height = height;
  }

  public int Volume
  {
    get
    {
      return length * length * length;
    }
  }
}

Compliant Solution

public class Cuboid
{
  private readonly int length;
  private int Breadth { get; set; } // Breadth is written but never read
  private readonly int height;  // height is written but never read

  public Cuboid(int length, int breadth, int height)
  {
    this.length = length;
    this.Breadth = breadth;
    this.height = height;
  }

  public int Volume
  {
    get
    {
      return length * Breadth * height;
    }
  }
}
